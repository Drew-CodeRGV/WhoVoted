All three district types use the same "r" number system:

| r365 | Precincts per district | PLANC2333_r365.xls | PLANS2168_r365.xls | PLANH2316_r365.xls |

## Integration Example

```python
import json
from pathlib import Path

# Load all district references
ref_dir = Path('data/district_reference')

with open(ref_dir / 'congressional_districts.json') as f:
    CONGRESSIONAL = json.load(f)

with open(ref_dir / 'state_senate_districts.json') as f:
    STATE_SENATE = json.load(f)

with open(ref_dir / 'state_house_districts.json') as f:
    STATE_HOUSE = json.load(f)

# Get district coverage
def get_coverage(district_type, district_num):
    """Calculate coverage for any district type."""
    
    refs = {
        'congressional': CONGRESSIONAL,
        'senate': STATE_SENATE,
        'house': STATE_HOUSE
    }
    
    reference = refs[district_type].get(str(district_num), {})
    total_counties = reference.get('total_counties', 0)
    
    # Query your database for actual data
    data_counties = query_database(district_type, district_num)
    
    return {
        'type': district_type,
        'district': district_num,
        'total_counties': total_counties,
        'counties_with_data': len(data_counties),
        'coverage_pct': len(data_counties) / total_counties * 100 if total_counties > 0 else 0
    }

# Example usage
print(get_coverage('congressional', 15))  # TX-15
print(get_coverage('senate', 27))         # SD-27
print(get_coverage('house', 40))          # HD-40
```

```python
def validate_upload(file_data, district_type, district_num):
    reference = get_reference(district_type, district_num)
    expected_counties = set(reference['counties'])
    uploaded_counties = set(file_data['county'].unique())
    
    invalid = uploaded_counties - expected_counties
    if invalid:
        return f"Invalid counties: {invalid}"
    return "Valid"
```
